圆角矩形公式
```css
  <path d="
    M x, y+ry
    a rx,ry 0 0 1 rx,-ry
    h w-2*rx
    a rx,ry 0 0 1 rx,ry
    v h-2*ry
    a rx,ry 0 0 1 -rx,ry
    h -(w-2*rx)
    a rx,ry 0 0 1 -rx,-ry
    z
  "/>

  <rect x="5" y="5" height="130" width="80" rx="2" ry="3"/>

  <path d="
    M 5 8
    a 2 3 0 0 1 2 -3
    h 76
    a 2 3 0 0 1 2 3
    v 124
    a 2 3 0 0 1 -2 3
    h -76
    a 2 3 0 0 1 -2 -3
    z 
  " />
```
